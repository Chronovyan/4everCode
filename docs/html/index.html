Install 4ever

# Install 4ever
curl -fsSL https://install.4ever.dev | sh

# Verify installation
4e --version

Your First Program

// hello.4e
import { time } from '4ever';

// Schedule a message for 5 seconds from now
at (time.now() + 5s) {
    console.log("Hello from the future!");
}

console.log("Scheduled a message for 5 seconds from now!");

Run It

# Run your program
4e run hello.4e
